4   import fi.maaretdufva.users.User;
5   import java.awt.event.KeyEvent;
6   import java.awt.event.KeyListener;
7   import java.util.logging.Level;
8   import java.util.logging.Logger;
9   import javax.swing.JLabel;
10  import javax.swing.JTextArea;
11  import robots.RobotTest;
12  
13  /*
14   * This is the action listener class. 
15   * <p>
16   * This class listens to user input and updates the window accordingly. It receives character information fron Game class and compares it to user input.
17   * It also sends information to the RobotTest class, when it is 
18   * used for testing purposes.
19   *
20   * @author Maaret Dufva
21   * @version 0.1.0
22   * @since   0.1.0
23   */
24  public class Listener implements KeyListener {
25  
26      private JTextArea writingArea;
27      private JTextArea text;
28      private JLabel points;
29      private JLabel level;
30      private Game game;
31      private User user;
32  
33      /**
34       * Constructor.
35       */
36      Listener(User user, JTextArea writingArea, JTextArea text, JLabel points, JLabel level) {
37          this.writingArea = writingArea;
38          this.text = text;
39          this.user = user;
40          this.game = new Game(user);
41          this.points = points;
42          this.level = level;
43      }
44  
45      @Override
46      public void keyTyped(KeyEvent ke) {
47  
48          if (ke.getKeyChar() == game.sendToListener()) {
49              user.addPoint();
50          } else {
51              user.deductPoint();
52          }
53  
54          //sendToRobot();
55          points.setText(String.valueOf(user.getPoints()));
56          level.setText(String.valueOf(user.countLevelbyPoints()));
57          writingArea.setText("");
58          text.setText(game.getTypeThis());
59      }
60  
61      /**
62       * If the RobotTest is being used, this method will provide it with
63       * information about the next character to type.
64       */
65      public void sendToRobot() {
66          //This is code for the the gui test robot. Unquote the command in keyTyped(), 
67          //and run the program from RobotTest class. 
68          //(Right click -> Run file or set RobotTest to Main class.)
69          //
70          RobotTest robot = new RobotTest();
71          try {
72              robot.setText(game.getKey());
73          } catch (Exception ex) {
74              Logger.getLogger(Listener.class.getName()).log(Level.SEVERE, null, ex);
75          }
76  
77      }
78  
79      @Override
80      public void keyPressed(KeyEvent ke) {
81          //Pressing enter doesn't deduct points
82          if (ke.getKeyCode() == KeyEvent.VK_ENTER) {
83              user.addPoint();
84          }
85      }
86  
87      @Override
88      public void keyReleased(KeyEvent ke) {
89      }
90  
91      public int getPoints() {
92          return user.getPoints();
93      }
94  
95  }
