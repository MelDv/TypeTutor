16  /**
17   * This class will deal with new and existing users.
18   * <p>
19   * If a user presses the Register-button, this class deals with getting user
20   * information, creating a new user and adding the user to the list of users. If
21   * a user wants to login, it finds their details and gives them to game class.
22   *
23   * @author Maaret Dufva
24   * @version 0.1.0
25   * @since 0.1.0
26   */
27  public class RegisterListener extends JPanel implements ActionListener {
28  
29      private User user;
30      private JButton register;
31      private JButton login;
32      private Game game;
33      private JFrame frame;
34      private JTextField giveUsername;
35      private JPasswordField givePassword;
36      private AllUsers all;
37  
38      /**
39       * Constructor.
40       *
41       * @param user from Gui
42       * @param game from Gui
43       * @param frame from Gui
44       * @param register from Gui
45       * @param giveUsername from Gui
46       * @param givePassword from Gui
47       * @param login from Gui
48       */
49      public RegisterListener(User user, Game game, JFrame frame, JButton register, JTextField giveUsername, JPasswordField givePassword, JButton login) {
50          this.register = register;
51          this.login = login;
52          this.game = game;
53          this.user = user;
54          this.frame = frame;
55          this.giveUsername = giveUsername;
56          this.givePassword = givePassword;
57          this.all = new AllUsers();
58      }
59  
60      @Override
61      public void actionPerformed(ActionEvent ae) {
62          String uname = giveUsername.getText();
63          char[] passw = givePassword.getPassword();
64  
65          if (ae.getSource() == register) {
66              if (register.getText().equals("Register")) {
67                  if (all.findUser(uname)) {
68                      badUsername("The username already in use. Pick another one");
69                  } else { //username fine
70                      user = new User(uname);
71                      user.setPassword(passw);
72                      all.addUser(user);
73                      game.setUser(user);
74                      register.setText("Logout");
75                      login.setEnabled(false);
76                      register.setEnabled(true);
77                      welcome();
78  
79                  }
80              } else if (register.getText().equals("Logout")) {
81                  frame.setTitle("Typetutor: Hello new user!");
82                  all.writeToFile(user);
83                  user = new User("temp");
84                  game.setUser(user);
85                  register.setText("Register");
86                  login.setEnabled(true);
87                  register.setEnabled(true);
88                  clearUp();
89              }
90          }
91          if (ae.getSource() == login) {
92              if (!all.findUser(uname)) {
93                  badUsername("User not found.");
94              } else if (user.testPassword(passw)) {
95                  welcome();
96                  login.setEnabled(false);
97                  register.setText("Logout");
98                  register.setEnabled(true);
99                  user = all.getUser(uname);
100                 game.setUser(user);
101             }
102         }
103     }
104 
105 //private methods for creating popup windows for error messages etc. and cleaning up the window.  
106     private void welcome() {
107         frame.setTitle("TypeTutor: Happy typing " + user.getUsername() + "!");
108         game.setUser(user);
109         clearUp();
110         welcomeDialog();
111     }
112 
113     private void welcomeDialog() {
114         JDialog jd = new JDialog();
115         JDialog.setDefaultLookAndFeelDecorated(true);
116         Object[] options = {"Cool!"};
117         String welcome = "Welcome " + user.getUsername() + "! Your score will now be saved. Don't forget your password!";
118         int response = JOptionPane.showOptionDialog(jd, welcome, "Welcome!",
119                 JOptionPane.PLAIN_MESSAGE, JOptionPane.INFORMATION_MESSAGE, null, options, options[0]);
120     }
121 
122     private void badUsername(String error) {
123         JDialog jd = new JDialog();
124         JDialog.setDefaultLookAndFeelDecorated(true);
125         Object[] options = {"Ok"};
126         int response = JOptionPane.showOptionDialog(jd, "Username not found", "Username",
127                 JOptionPane.WARNING_MESSAGE, JOptionPane.WARNING_MESSAGE, null, options, options[0]);
128 
129         if (response == JOptionPane.YES_OPTION) {
130             jd.dispose();
131         }
132     }
133 
134     private void clearUp() {
135         all.writeToFile(user);
136         this.giveUsername.setText("");
137         this.givePassword.setText("");
138         frame.validate();
139         frame.repaint();
140     }
