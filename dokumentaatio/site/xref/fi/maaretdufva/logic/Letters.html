61       * be typed.
62       */
63      public String determineString(int givenLevel) {
64          determineLevel(givenLevel);
65  
66          this.letters = letterSequences[level];
67  
68          if (level < 4) {
69              randomString();
70          } else if (level < 9) {
71              randomStringWithCapitals();
72          } else if (level < 16) {
73              randomStringWithEndPunctuation();
74          } else {
75              randomStringWithPunctuation();
76          }
77  
78          String stringOfWords = "";
79          for (int i = 0; i < words.length; i++) {
80              stringOfWords += words[i];
81          }
82  
83          return stringOfWords;
84      }

92      public int determineLevel(int givenLevel) {
93          if (givenLevel <= 0) {
94              this.level = 0;

96              this.level = letterSequences.length - 1;
97          } else {
98              this.level = givenLevel;
99          }

119     /**
120      * Creates a random Sring using characters given as a String parameter. Each
121      * random word has the number of characters given as an integer parameter.
122      *
123      * @return a String table of ten random character sequences.
124      */
125     public String[] randomString() {
126         Random r = new Random();
127         int wordLength = 0;
128 
129         for (int i = 0; i < words.length; i++) {
130             String typeThis = "";
131             if (this.letters.length() < 5) {
132                 wordLength = 3;
133             } else {
134                 wordLength = r.nextInt(3) + 6;
135             }
136             for (int j = 0; j < wordLength; j++) {
137                 char letter = (this.letters.charAt(r.nextInt(this.letters.length())));
138                 typeThis = typeThis + letter;
139             }
140             if (i > 0) {
141                 words[i] = " " + typeThis;
142             } else {
143                 words[i] = typeThis;
144             }
145         }
146         return words;
147     }

149     private String[] randomStringWithCapitals() {
150         randomString();
151 
152         for (int i = 0; i < words.length; i++) {
153             if (i % 3 == 0) {
154                 String temp = words[i];
155                 temp = temp.trim();
156                 temp = " " + temp.substring(0, 1).toUpperCase() + temp.substring(1);
157                 words[i] = temp;
158             }
159         }
160         return words;
161     }

163     private String[] randomStringWithEndPunctuation() {
164         Random r = new Random();
165         String[] punctuation = {".", "!", "?"};
166 
167         randomString();
168         randomStringWithCapitals();
169 
170         for (int i = 0; i < words.length; i++) {
171             if (i != 0 && i % 2 == 0) {
172                 String temp = words[i];
173                 temp = temp + punctuation[r.nextInt(punctuation.length)];
174                 words[i] = temp;
175             }
176         }
177         return words;
178     }
