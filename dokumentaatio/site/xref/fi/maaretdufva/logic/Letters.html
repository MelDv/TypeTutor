151         int wordLength = 0;
152         String typeThis = "";
153 
154         if (numberOfLetters == 0) {
155             wordLength = r.nextInt(3) + 6;
156         } else {
157             wordLength = numberOfLetters;
158         }
159 
160         for (int i = 0; i < numberOfWords; i++) {
161             for (int j = 0; j < wordLength; j++) {
162                 char letter = (charactersToLearn.charAt(r.nextInt(charactersToLearn.length())));
