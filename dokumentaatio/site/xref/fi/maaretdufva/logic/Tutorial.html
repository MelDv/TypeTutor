30      /**
31       * Guides for typing each letter.
32       *
33       * @param level Game class calls for this method and provides level
34       * information.
35       * @return instructions for typing efficiently based on level.
36       */
37      public String letterTutorials(int level) {
38          if (level == 0) {
39              return "Nakuta äffää ja jiitä etusormilla.";
40          }
41          if (level == 1) {
42              return "Nakuta keskisormilla deetä ja koota.";
43          }
44          if (level == 2) {
45              return "Loistavaa! Nyt f, j, d ja k yhdessä.";
46          }
47          if (level == 3) {
48              return "Harjoittele ällää ja ässää nimettömillä.";
49          }
50          if (level == 4) {
51              return "Hyvin menee! Nyt jfkdls kaikki yhdessä.";
52          }
53          if (level == 5) {
54              return "Otetaan pikkurilli mukaan. Harjoittele aa ja öö.";
55          }
56          if (level == 6) {
57              return "Harjoittele vielä yhdistämään lsöajf.";
58          }
59          if (level == 7) {
60              return "Muistatko kaikki keskirivin näppäimet? Harjoittele jfkdlsöa.";
61          }
62          if (level == 8) {
63              return "Edistyt erinomaisesti! Siirrytään yläriville. "
64                      + "Kirjoita kirjaimia är ja uu etusormilla.";
65          }
66          return "";
67          //To be continued
68      }
