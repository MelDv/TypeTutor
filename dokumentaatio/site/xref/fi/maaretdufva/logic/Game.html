5   package fi.maaretdufva.logic;
6   
7   import fi.maaretdufva.users.AllUsers;
8   import fi.maaretdufva.users.User;
9   
10  public class Game {
11  
12      private User user;
13      private AllUsers all;
14      private String typeThis;
15      private int level;
16  
17      public Game(User givenUser) {
18          this.user = givenUser;
19          this.all = new AllUsers(user);
20          this.typeThis = "";
21          this.level = 0;
22          System.out.println(user.getPoints());
23      }
24  
25      public Game() {
26          start();
27      }
28  
29      public User myUser() {
30          return this.user;
31      }
32  
33      public void start() {
34          if (user == null || user.getPoints() == 0) {
35              OnBoarding begin = new OnBoarding();
36              begin.start();
37          } else {
38              determineLevel();
39              System.out.println(level);
40              determineTypeThis();
41              System.out.println(typeThis);
42              sendToListener();
43          }
44      }
45  
46      public int determineLevel() {
47          if (this.level != user.getLevel()) {
48              this.level = user.getLevel();
49              Tutorial tut = new Tutorial();
50              tut.letterTutorials(level);
51          }
52          return this.level;
53      }
54  
55      public void determineTypeThis() {
56          Letters letters = new Letters();
57          typeThis = letters.determineString(level);
58      }
59  
60      public char sendToListener() {
61          if (typeThis == null || typeThis.length() == 0) {
62              determineLevel();
63              determineTypeThis();
64          }
65  
66          char typeLetter = typeThis.charAt(0);
67          typeThis = typeThis.substring(1);
68          return typeLetter;
69      }
70  
71      public String getTypeThis() {
72          return typeThis;
73      }
74  }
